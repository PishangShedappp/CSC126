#include <iostream>

using namespace std;

int main() {
	double sideB, rad, square, shadedRegion;
	
	cout << "Side B: ";
	cin >> sideB;
	cout << "Radius: ";
	cin >> rad;
	
	square = sideB * sideB;
	shadedRegion = square - (4*(3.142*(rad*rad)));
	
	cout << "Area of the square: " << square << endl;
	cout << "Shaded Region: " << shadedRegion;
}
